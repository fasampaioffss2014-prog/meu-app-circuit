function processarTextoExtraido(texto) { const linhas = texto.split('\n'); const padraoCep = /\b\d{5}-?\d{3}\b/;; let contador = proximoId; linhas.forEach((linha, index) => { linha = linha.trim(); if (!linha) return; // Se a linha contiver um CEP ou a linha anterior/posterior indicar endereço const matchCep = inlineMatch(linha, padraoCep) || linha.includes("20000"); // detecta CEPs comuns mesmo sem traço const ehLinhaDeRua = /rua|av\.|avenida|alameda|travessa|pça|praça/i.test(linha); if (ehLinhaDeRua || matchCep) { // Se achou uma rua mas o CEP está na linha de baixo, tenta juntar let enderecoCompleto = linha; let cepDetectado = "20000000"; // Padrão caso falhe const mCep = inlineMatch(linha, /\b\d{5}-?\d{3}\b/); if (mCep) { cepDetectado = mCep[0].replace('-', ''); enderecoCompleto = linha.replace(mCep[0], '').trim(); } // Limpa caracteres estranhos que o OCR gera no reflexo enderecoCompleto = enderecoCompleto.replace(/^[,.\s\-|]+|[,.\s\-|]+$/g, '').replace(/[\u00C0-\u00FF]/g, (c) => c).trim(); if (enderecoCompleto.length > 5 && !enderecoCompleto.toLowerCase().includes("remetente")) { const novaRota = { Name: contador++, Address: enderecoCompleto, PostalCode: cepDetectado }; rotasExtraidas.push(novaRota); adicionarLinhaNaTabela(novaRota); } } }); if (rotasExtraidas.length > 0) { statusDiv.textContent = `${rotasExtraidas.length} rota(s) carregada(s). Corrija se necessário!`; btnExportar.disabled = false; } else { statusDiv.textContent = "Erro na leitura. Tente usar um print direto da tela ou mude o ângulo da foto para tirar o reflexo."; } }
